40 DLL_PUBLIC inline double
41 signedVolumeFromFace(gcs::Face &f, gcs::VertexPositionGeometry &vpg, gc::Vector3 center) {
42  gc::Vector3 p[3];
43  size_t i = 0;
44  for (gcs::Vertex v : f.adjacentVertices()) {
45  p[i] = vpg.inputVertexPositions[v] - center;
46  i++;
47  }
48  double v321 = p[2].x * p[1].y * p[0].z;
49  double v231 = p[1].x * p[2].y * p[0].z;
50  double v312 = p[2].x * p[0].y * p[1].z;
51  double v132 = p[0].x * p[2].y * p[1].z;
52  double v213 = p[1].x * p[0].y * p[2].z;
53  double v123 = p[0].x * p[1].y * p[2].z;
54 
55  return (-v321 + v231 + v312 - v132 - v213 + v123) / 6.0;
56 }

91 DLL_PUBLIC inline Eigen::Matrix<double, Eigen::Dynamic, 3>
92 rowwiseCrossProduct(Eigen::Matrix<double, Eigen::Dynamic, 3> A,
93  Eigen::Matrix<double, Eigen::Dynamic, 3> B) {
94  Eigen::Matrix<double, Eigen::Dynamic, 3> C;
95  if (A.rows() != B.rows()) {
96  throw std::runtime_error("The input matrices must have same sizes!");
97  return C;
98  } else {
99  C.resize(A.rows(), 3);
100  for (size_t i = 0; i < A.rows(); i++) {
101  C.row(i) = A.row(i).cross(B.row(i));
102  }
103  return C;
104  }
105 }

169 DLL_PUBLIC inline void
170 vertexShift(gcs::SurfaceMesh &mesh, gcs::VertexPositionGeometry &vpg,
171  Eigen::Matrix<bool, Eigen::Dynamic, 1> mask) {
172 
173  for (gcs::Vertex v : mesh.vertices()) {
174 
175  if (mask(vpg.vertexIndices[v])) {
176 
177  gc::Vector3 baryCenter{0.0, 0.0, 0.0};
178  double n_vAdj = 0.0;
179  for (gcs::Vertex vAdj : v.adjacentVertices()) {
180  baryCenter += vpg.inputVertexPositions[vAdj];
181  n_vAdj += 1.0;
182  }
183  baryCenter /= n_vAdj;
184  for (gcs::Halfedge he : v.outgoingHalfedges()) {
185  gcs::Halfedge base_he = he.next();
186  vpg.inputVertexPositions[v] =
187  baryCenter - gc::dot(vpg.vertexNormals[v],
188  baryCenter - vpg.inputVertexPositions[v]) *
189  vpg.vertexNormals[v];
190  }
191  }
192  }
193 }

262 DLL_PUBLIC inline Eigen::Matrix<std::uint32_t, Eigen::Dynamic, 3, Eigen::RowMajor>
263 getFaceVertexMatrix(gcs::SurfaceMesh &mesh) {
264  if (!mesh.isTriangular())
265  throw std::runtime_error("Mesh is not triangular.");
266 
267  Eigen::Matrix<std::uint32_t, Eigen::Dynamic, 3, Eigen::RowMajor> result(mesh.nFaces(), 3);
268 
269  gcs::VertexData<std::size_t> vInd = mesh.getVertexIndices();
270 
271  std::size_t i = 0;
272  for (gcs::Face f : mesh.faces()) {
273  std::uint32_t a, b, c;
274  gcs::Halfedge tmp = f.halfedge();
275  a = vInd[tmp.vertex()];
276  tmp = tmp.next();
277  b = vInd[tmp.vertex()];
278  tmp = tmp.next();
279  c = vInd[tmp.vertex()];
280 
281  result.row(i++) << a, b, c;
282  }
283  return result;
284 }
