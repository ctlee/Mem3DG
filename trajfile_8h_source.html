303  TrajFile(const std::string &filename, gcs::SurfaceMesh &mesh,
304  gcs::VertexPositionGeometry &refVpg, const NcFile::FileMode fMode)
305  : filename(filename), // fd(new NcFile(filename, fMode)),
306  writeable(true) {
307 
308  fd = new NcFile(filename, fMode);
309  // initialize data
310  fd->putAtt(CONVENTIONS_NAME, CONVENTIONS_VALUE);
311  fd->putAtt(CONVENTIONS_VERSION_NAME, CONVENTIONS_VERSION_VALUE);
312 
313  frame_dim = fd->addDim(FRAME_NAME);
314  npolygons_dim = fd->addDim(NPOLYGONS_NAME, mesh.nFaces());
315  nvertices_dim = fd->addDim(NVERTICES_NAME, mesh.nVertices());
316  spatial_dim = fd->addDim(SPATIAL_DIMS_NAME, SPATIAL_DIMS);
317  polygon_order_dim = fd->addDim(POLYGON_ORDER_NAME, POLYGON_ORDER);
318 
319  // Initialize topology data block
320  topology =
321  fd->addVar(TOPO_VAR, nc::ncUint, {npolygons_dim, polygon_order_dim});
322 
323  // Populate topology data
324  Eigen::Matrix<std::uint32_t, Eigen::Dynamic, 3, Eigen::RowMajor>
325  faceMatrix = getFaceVertexMatrix(mesh);
326  std::uint32_t *topodata = faceMatrix.data();
327  topology.putVar(topodata);
328 
329  // Initialize reference coordinate data block
330  refcoord = fd->addVar(REFCOORD_VAR, netCDF::ncDouble,
331  {nvertices_dim, spatial_dim});
332 
333  // Populate reference coordinate data
334  double *refcoorddata;
335  refcoorddata = gc::EigenMap<double, 3>(refVpg.inputVertexPositions).data();
336  refcoord.putVar(refcoorddata);
337 
338  time_var = fd->addVar(TIME_VAR, netCDF::ncDouble, {frame_dim});
339  time_var.putAtt(UNITS, TIME_UNITS);
340 
341  coord_var = fd->addVar(COORD_VAR, netCDF::ncDouble,
342  {frame_dim, nvertices_dim, spatial_dim});
343  coord_var.putAtt(UNITS, LEN_UNITS);
344 
345  vel_var = fd->addVar(VEL_VAR, netCDF::ncDouble,
346  {frame_dim, nvertices_dim, spatial_dim});
347 
348  meancurve_var =
349  fd->addVar(MEANCURVE_VAR, netCDF::ncDouble, {frame_dim, nvertices_dim});
350 
351  sponcurve_var =
352  fd->addVar(SPONCURVE_VAR, netCDF::ncDouble, {frame_dim, nvertices_dim});
353 
354  externpress_var = fd->addVar(EXTERNPRESS_VAR, netCDF::ncDouble,
